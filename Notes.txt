            IConfigLanguage activeLang = editTimeEditor.Parser.Formats as IConfigLanguage;
            ConfigurationDialog langConfig = new ConfigurationDialog(editTimeEditor.Configurator, activeLang);
            if (langConfig.ShowDialog(this) == DialogResult.OK && activeLang != null)
            {
                IConfigLanguage newLang = langConfig.Configurator.KnownLanguageNames.Contains(activeLang.Language) ? langConfig.Configurator[activeLang.Language] : langConfig.Configurator.DefaultLanguage;
                if (newLang != null)
                {
                    editTimeEditor.Configurator = langConfig.Configurator;
                    editTimeEditor.ApplyConfiguration(newLang);
                }
            }
